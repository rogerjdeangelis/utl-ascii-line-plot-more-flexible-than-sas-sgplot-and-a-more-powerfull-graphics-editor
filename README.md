    proc sgplot data=have;
    scatter x=x y=y
     /markerchar=year
      markercharattrs
        =(size=10 )
      labelstrip;
    xaxis
     values=
      (-90 to 90 by 10)
      min=-90 max=90
      valueshint;
    yaxis display=none
      offsetmin=0
      offsetmax=0.1
      min=1.6
      valueshint;
    run;
